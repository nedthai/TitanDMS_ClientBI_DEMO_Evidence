```sql monthly_sales
WITH raw_monthly AS (
  SELECT
    DATE_TRUNC('month', Invoice_Date) as report_month,
    COUNT(Vehicle_Key) as actual_sales
  FROM demo_dealer_ai.vehicle_sales
  WHERE Invoice_Date < '2026-03-01'
  GROUP BY DATE_TRUNC('month', Invoice_Date)
),
numbered_months AS (
  SELECT
    report_month,
    actual_sales,
    ROW_NUMBER() OVER (ORDER BY report_month) as x_val
  FROM raw_monthly
),
regression_stats AS (
  SELECT
    regr_slope(actual_sales, x_val) as slope,
    regr_intercept(actual_sales, x_val) as intercept,
    regr_r2(actual_sales, x_val) as r_squared,
    corr(actual_sales, x_val) as correlation_coeff,
    MAX(x_val) as max_x
  FROM numbered_months
),
future_months AS (
  SELECT 1 as step UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL 
  SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL 
  SELECT 9 UNION ALL SELECT 10
),
forecast_data AS (
  SELECT DISTINCT
    (SELECT MAX(report_month) FROM numbered_months) + (INTERVAL 1 MONTH * f.step) as report_month,
    NULL as actual_sales,
    (s.slope * (s.max_x + f.step) + s.intercept) as predicted_sales,
    'Forecast' as data_type
  FROM future_months f
  CROSS JOIN regression_stats s
),
historical_with_predictions AS (
  SELECT
    n.report_month,
    n.actual_sales,
    (s.slope * n.x_val + s.intercept) as predicted_sales,
    'Historical' as data_type
  FROM numbered_months n
  CROSS JOIN regression_stats s
)
SELECT * FROM historical_with_predictions
UNION ALL
SELECT * FROM forecast_data
ORDER BY report_month
```

```sql forecast_metrics
WITH raw_monthly AS (
  SELECT
    DATE_TRUNC('month', Invoice_Date) as report_month,
    COUNT(Vehicle_Key) as actual_sales
  FROM demo_dealer_ai.vehicle_sales
  GROUP BY DATE_TRUNC('month', Invoice_Date)
),
numbered_months AS (
  SELECT
    report_month,
    actual_sales,
    ROW_NUMBER() OVER (ORDER BY report_month) as x_val
  FROM raw_monthly
)
SELECT
  regr_slope(actual_sales, x_val) as slope,
  regr_intercept(actual_sales, x_val) as intercept,
  regr_r2(actual_sales, x_val) as r_squared,
  corr(actual_sales, x_val) as correlation_coeff
FROM numbered_months
```

        <p>
            To predict future vehicle sales, we utilize a foundational statistical method known as <strong>Simple Linear Regression</strong>. 
            This technique models the relationship between time (months) and sales volume by fitting a linear equation to observed data. By minimizing the sum of the squares of the vertical deviations from the data points to the line—a principle called <strong>Ordinary Least Squares (OLS)</strong>—we find the line of best fit.
        </p>
        <p>
            The mathematical model generated by DuckDB is represented by the formula <code>y = mx + b</code>:
        </p>
        <ul class="list-disc pl-5 mt-2 mb-4">
            <li><strong>y (Predicted Sales)</strong>: The value we are forecasting.</li>
            <li><strong>m (Slope = <Value data={forecast_metrics} column=slope fmt="num2" />)</strong>: The change in sales per month, indicating the upward or downward trend.</li>
            <li><strong>x (Time)</strong>: The sequential index of the month.</li>
            <li><strong>b (Intercept = <Value data={forecast_metrics} column=intercept fmt="num2" />)</strong>: The theoretically predicted sales at the baseline starting point.</li>
        </ul>
        <p>
            The reliability of this model is established using two key metrics: the <strong>R-squared</strong> value, which indicates how much variance in sales is explained by time, and the <strong>Pearson Correlation Coefficient</strong>, which measures the strength of the linear trend.
        </p>
